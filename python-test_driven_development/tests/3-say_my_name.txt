def say_my_name(first_name, last_name=None):
    """Function to print a person's name.

    Args:
        first_name (str): The first name of the person.
        last_name (str, optional): The last name of the person. Defaults to None.

    Raises:
        TypeError: If first_name is not a string.
        TypeError: If last_name is provided and is not a string.
    """
    if not isinstance(first_name, str):
        raise TypeError("first_name must be a string")

    if last_name is not None and not isinstance(last_name, str):
        raise TypeError("last_name must be a string")

    if last_name:
        print(f"My name is {first_name} {last_name}")
    else:
        print(f"My name is {first_name}")

# Test cases
try:
    say_my_name("John", "Smith")
    say_my_name("Walter", "White")
    say_my_name("Bob")
    say_my_name(12, "White")
    say_my_name(None)
    say_my_name(None , "poppins")
    say_my_name("mary" , None)
    say_my_name(1, "Poppins")
    say_my_name("mary", 2)
    say_my_name(["mary"], "poppins")
    say_my_name("mary", ["poppins"])
    say_my_name()
except TypeError as e:
    print(e)
